from tkinter import
class FoodSecurifyApp(Tk)
def __init__(self)
super().__init__()
self.title(Food Securify)
self.geometry(800x600)
self.configure(bg='white')
self.frames = {}
for F in (HomePage, PlacesPage, DishesPage, CaloriesPage,
WaterPage)
page_name = F.__name__
frame = F(parent=self, controller=self)
self.frames[page_name] = frame
frame.grid(row=0, column=0, sticky=nsew)
self.show_frame(HomePage)
self.center_window()
def show_frame(self, page_name)
frame = self.frames[page_name]
frame.tkraise()
def center_window(self)
self.update_idletasks()
width = self.winfo_width()
height = self.winfo_height()
screen_width = self.winfo_screenwidth()
screen_height = self.winfo_screenheight()
x = (screen_width - width) 2
y = (screen_height - height) 2
self.geometry(f{width}x{height}+{x}+{y})
class HomePage(Frame)

title_label = Label(self, text=Food Securify, font=(Times New

title_label.pack(pady=20)

text_frame.pack(expand=True, fill=BOTH, padx=20)

scrollbar.pack(side=RIGHT, fill=Y)

def create_page_layout(frame, title, description, controller)
Label(frame, text=title, font=(Arial, 22, bold),
bg=frame[bg]).pack(pady=(30, 10))
Label(frame, text=description, font=(Arial, 14), bg=frame[bg],
wraplength=700, justify=LEFT).pack(pady=(0, 40))
back_frame = Frame(frame, bg=frame[bg])
back_frame.pack(side=BOTTOM, fill=X, pady=20, padx=20)
Button(back_frame, text=⬅ Back, command=lambda
controller.show_frame(HomePage)).pack(anchor=w)
